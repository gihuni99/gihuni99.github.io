        down=0

        up=0
        down=0
    if down>2 and S[0]>=c:
        S[1]+=S[0]//c
        S[0]=S[0]%c
    elif up>2:
        S[0]+=S[1]*c
        S[1]=0
if J[0]+J[1]*chart[-1]>S[0]+S[1]*chart[-1]:
    print('BNP')

    print('TIMING')

    print('SAMESAME')
