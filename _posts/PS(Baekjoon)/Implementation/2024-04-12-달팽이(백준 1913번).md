            pre_num=my_map[x][y]
            x=x+dx[3]
            y=y+dy[3]
            my_map[x][y]=pre_num+1
            for _ in range(length-1):
                pre_num=my_map[x][y]
                x=x+dx[i]
                y=y+dy[i]
                my_map[x][y]=pre_num+1
        else:
            for _ in range(length):
                pre_num=my_map[x][y]
                x=x+dx[i]
                y=y+dy[i]
                my_map[x][y]=pre_num+1
    length+=2
a_x=0
a_y=0
for i in range(N):
    for j in range(N):
        print(my_map[i][j],end=' ')
        if my_map[i][j]==number:
            a_x=i+1
            a_y=j+1
    print('')
print(a_x,a_y)
