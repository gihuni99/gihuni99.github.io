import sys
from collections import defaultdict, deque

def DFS(graph,i,visited):
    visited[i]=True
    print(i,end=' ')
    for g in graph[i]:
        if not visited[g]:
            DFS(graph,g,visited)
        
def BFS(graph,i,visited):
    queue=deque()
    queue.append(i)
    visited[i]=True
    while queue:
        v=queue.popleft()
        print(v,end=' ')
        for g in graph[v]:
            if not visited[g]:
                queue.append(g)
                visited[g]=True

N,M,V=map(int,input().split())

for _ in range(M):
    n1,n2=map(int,sys.stdin.readline().strip().split())
    graph[n1].append(n2)
    graph[n2].append(n1)
for key in graph:
    graph[key].sort()
    
visited=[False]*(N+1)
DFS(graph,V,visited)
print('')
