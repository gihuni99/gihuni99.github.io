```python
import sys
from collections import defaultdict, deque

def BFS(graph,i,visited):
    queue=deque()
    parent=[[] for _ in range(len(visited))]
    visited[i]=True
    queue.append(i)
    while queue:
        v=queue.popleft()
        for g in graph[v]:
            if not visited[g]:
                visited[g]=True
                queue.append(g)
                parent[g].append(v)
    return parent
N=int(input())

graph=defaultdict(list)
visited=[False]*(N+1)
for _ in range(N-1):
    n1,n2=map(int,sys.stdin.readline().strip().split())
    graph[n1].append(n2)
    graph[n2].append(n1)
answer=BFS(graph,1,visited)
for i in range(2,len(answer)):
    print(answer[i][0])
```
