import sys
from collections import deque

def BFS(my_map,x,y,visited,n,m):
    dx=[-1,1,0,0]
    dy=[0,0,-1,1]
    visited[x][y]=True
    my_map[x][y]=0
    queue=deque()
    queue.append((x,y))
    while queue:
        v=queue.popleft()
        for i in range(4):
            mx=v[0]+dx[i]
            my=v[1]+dy[i]
            if 0<=mx<n and 0<=my<m and not visited[mx][my] and my_map[mx][my]:
                visited[mx][my]=True
                queue.append((mx,my))
                my_map[mx][my]=my_map[v[0]][v[1]]+1
    return my_map

input=sys.stdin.readline

my_map=[]
visited=[[False]*m for _ in range(n)]
for n_i in range(n):
    l=list(map(int,input().strip().split()))
    my_map.append(l)
x=0
y=0
for i in range(n):
    stop=0
    for j in range(m):
        if my_map[i][j]==2:
            x=i
            y=j
            stop=1
            break
    if stop==1:
        break
my_map=BFS(my_map,x,y,visited,n,m)
for i in range(n):
    for j in range(m):
        if my_map[i][j]==1 and not visited[i][j]:
            print(-1,end=' ')
