import sys

vowel='aeiou'
while True:
    ID=sys.stdin.readline().strip()
    if ID=='end':
        break
    accept=False
    v_conti=0
    c_conti=0
    v_count=0
    for i,w in enumerate(ID):
        if w in vowel:
            v_conti+=1
            v_count+=1
            c_conti=0
        else:
            c_conti+=1
            v_conti=0
        if i==0:
            continue
        if w!='e' and w!='o' and w==ID[i-1]:
            break
        elif v_conti>2 or c_conti>2:
            break
    else:
        if v_count!=0:
            accept=True
    if accept:
        print(f"<{ID}> is acceptable.")
    else:
        print(f"<{ID}> is not acceptable.")
